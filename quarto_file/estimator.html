Product Kernel Estimator

\[ \hat f_{X,Y}(x,y) = \frac{1}{n h_x h_y} \sum_{i=1}^{n} K\!\left(\frac{x-X_i}{h_x}\right) K\!\left(\frac{y-Y_i}{h_y}\right) \]

set.seed(123)

n <- 500
x <- runif(n, -3, 3)

# Heteroskedastic noise
sigma <- 0.3 + 0.2 * abs(x)
y <- sin(x) + rnorm(n, 0, sigma)

# Define kernel and bandwidths
K <- function(u) dnorm(u)

hx <- 0.4
hy <- 0.3

# Choose evaluation grid
x0 <- 1 # Fix X = 1
y_grid <- seq(-2, 2, length.out = 200)

# Compute conditional density at X = 1
cond_density <- sapply(y_grid, function(y0) {
    weights_x <- K((x0 - x) / hx)
    numerator <- sum(weights_x * K((y0 - y) / hy))
    denominator <- sum(weights_x)
    numerator / (hy * denominator)
})

# Plot Conditional Density at X = 1
plot(y_grid, cond_density,
    type = "l",
    lwd = 3,
    main = "Estimated Conditional Density at X = 1",
    xlab = "y",
    ylab = "f(y | X = 1)"
)
# To visualize the full conditional density surface:
x_grid <- seq(-3, 3, length.out = 50)
y_grid <- seq(-2, 2, length.out = 100)

density_matrix <- outer(x_grid, y_grid, Vectorize(function(x0, y0) {
    weights_x <- K((x0 - x) / hx)
    numerator <- sum(weights_x * K((y0 - y) / hy))
    denominator <- sum(weights_x)
    numerator / (hy * denominator)
}))

# Plot surface:
persp(x_grid, y_grid, density_matrix,
    theta = 30, phi = 30,
    xlab = "X",
    ylab = "Y",
    zlab = "f(Y|X)"
)

library(betareg)

set.seed(123)

n <- 1000 # sample size
mu_beta <- c(0, 0, 0) # means of X1, X2, X3

# Covariance matrix for correlated covariates
Sigma <- matrix(c(
    1, 0.5, 0.3,
    0.5, 1, 0.4,
    0.3, 0.4, 1
), nrow = 3)

# Generate correlated covariates
X <- mvrnorm(n, mu = mu_beta, Sigma = Sigma)
X1 <- X[, 1]
X2 <- X[, 2]
X3 <- X[, 3]

# simulate response
phi <- 5 # precision parameter
beta <- c(0.5, 0.7, -0.3, 0.2) # beta0, beta1, beta2, beta3

# Linear predictor
eta <- beta[1] + beta[2] * X1 + beta[3] * X2 + beta[4] * X3

# Inverse logit to get mean in (0,1)
mu <- plogis(eta)

# Simulate beta response
Y <- rbeta(n, shape1 = mu * phi, shape2 = (1 - mu) * phi)


fit <- betareg(Y ~ X1 + X2 + X3, link = "logit")
