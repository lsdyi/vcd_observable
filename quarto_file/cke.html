# 2. Generate Synthetic Multivariate Data
set.seed(123)
n <- 300

# Continuous predictor
x1 <- runif(n, -2, 2)

# Categorical predictor (e.g., 'Group A' or 'Group B')
x2 <- factor(sample(c("A", "B"), n, replace = TRUE))

# Generate a Count Response (y)
# We'll use a Poisson-like structure where the mean changes with x1 and x2
lambda <- exp(0.5 + 1.2 * sin(x1) + as.numeric(x2 == "B") * 0.5)
y_numeric <- rpois(n, lambda)

# CRITICAL STEP: Convert count to an ORDERED FACTOR for discrete kernel estimation
y_count <- ordered(y_numeric)

my_data <- data.frame(y_count, x1, x2)

# Sample data (two clusters)
x <- c(1.8, 2.0, 2.1, 2.2, 4.8, 5.0, 5.1, 5.2)

# Create grid of points to evaluate density
x_grid <- seq(1, 6, length.out = 200)

# -----------------------------
# 1. Histogram (density)
# -----------------------------
hist_info <- hist(x, breaks = seq(1, 6, by = 1), plot = FALSE)

# Convert histogram to density values
hist_density <- hist_info$density
hist_mids <- hist_info$mids

# -----------------------------
# 2. Naive estimator function
# -----------------------------
naive_density <- function(x_grid, data, h) {
    n <- length(data)
    sapply(x_grid, function(x0) {
        sum(abs(data - x0) < h / 2) / (n * h)
    })
}

# Compute naive estimator
h <- 1
naive_vals <- naive_density(x_grid, x, h)

# -----------------------------
# Plot
# -----------------------------
plot(x_grid, naive_vals,
    type = "l", lwd = 2,
    xlab = "x", ylab = "Density",
    main = "Histogram vs Naive Density Estimator"
)

# Add histogram as step function
for (i in 1:length(hist_mids)) {
    segments(hist_info$breaks[i], hist_density[i],
        hist_info$breaks[i + 1], hist_density[i],
        lwd = 2, lty = 2
    )
}

legend("topright",
    legend = c("Naive estimator", "Histogram"),
    lty = c(1, 2),
    lwd = 2
)
